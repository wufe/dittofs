// ErrRemoteUnavailable is returned when a remote store operation is needed
// but the remote store is currently unreachable. Protocol handlers should
// map this to appropriate I/O error codes (NFS3ERR_IO, NFS4ERR_IO,
// STATUS_UNEXPECTED_IO_ERROR).

// The error is intentionally returned early (before attempting network I/O)

// logOfflineRead logs a read failure due to remote unavailability.

func (m *Syncer) logOfflineRead(method, payloadID string, blockIdx uint64) {

            "payloadID", payloadID,
            "blockIdx", blockIdx,
            "outage_duration", m.RemoteOutageDuration().Truncate(time.Second))

        logger.Debug("Read blocked: remote store unavailable",
            "method", method,
            "payloadID", payloadID,
            "blockIdx", blockIdx)

}
