           const gib = 1024 * 1024 * 1024
           const mib = 1024 * 1024
           if n >= gib {
               return fmt.Sprintf("%.1fGiB", float64(n)/float64(gib))
           }
           return fmt.Sprintf("%.0fMiB", float64(n)/float64(mib))
