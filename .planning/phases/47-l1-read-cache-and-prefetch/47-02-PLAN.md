    ```go
    func (bs *BlockStore) readAtInternal(ctx context.Context, payloadID, cowSource string, data []byte, offset uint64) (int, error) {
        if len(data) == 0 {
            return 0, nil
        }

        // L1 fast path: try to serve entirely from L1 cache
        if bs.readCache != nil && cowSource == "" {
            startBlock := offset / blockstore.BlockSize
            endBlock := (offset + uint64(len(data)) - 1) / blockstore.BlockSize
            allHit := true
            for blockIdx := startBlock; blockIdx <= endBlock; blockIdx++ {
                blockStart := blockIdx * blockstore.BlockSize
                blockOff := uint32(0)
                if offset > blockStart {
                    blockOff = uint32(offset - blockStart)
                }
                destOff := uint64(0)
                if blockStart > offset {
                    destOff = blockStart - offset
                }
                remaining := uint64(len(data)) - destOff
                if remaining <= 0 {
                    break
                }
                _, hit := bs.readCache.Get(payloadID, blockIdx, data[destOff:], blockOff)
                if !hit {
                    allHit = false
                    break
                }
            }
            if allHit {
                // Notify prefetcher
                if bs.prefetcher != nil {
                    endBlock := (offset + uint64(len(data)) - 1) / blockstore.BlockSize
                    bs.prefetcher.OnRead(payloadID, endBlock)
                }
                return len(data), nil
            }
        }

        // Existing read path (local cache -> syncer download)
        // ... keep existing code ...

        // After successful read, fill L1 with full blocks
        if bs.readCache != nil && cowSource == "" {
            bs.fillL1FromRead(ctx, payloadID, offset, uint64(len(data)))
        }

        // Notify prefetcher
        if bs.prefetcher != nil && cowSource == "" {
            endBlock := (offset + uint64(len(data)) - 1) / blockstore.BlockSize
            bs.prefetcher.OnRead(payloadID, endBlock)
        }

        return len(data), nil
    }
    ```

    Add `fillL1FromRead` helper:

    ```go
    func (bs *BlockStore) fillL1FromRead(ctx context.Context, payloadID string, offset, length uint64) {
        startBlock := offset / blockstore.BlockSize
        endBlock := (offset + length - 1) / blockstore.BlockSize
        for blockIdx := startBlock; blockIdx <= endBlock; blockIdx++ {
            if bs.readCache.Contains(payloadID, blockIdx) {
                continue // already cached

            data, dataSize, err := bs.local.GetBlockData(ctx, payloadID, blockIdx)
            if err == nil && data != nil {
                bs.readCache.Put(payloadID, blockIdx, data, dataSize)
            }
        }
    }
    ```

    ```go
    func (bs *BlockStore) autoPromoteAfterFlush(ctx context.Context, payloadID string) {
        // Read all blocks for this file from local store and populate L1.
        // This is effectively free since the data was just flushed and is in OS page cache.
        files := bs.local.ListFiles()
        for _, f := range files {
            if f != payloadID {
                continue
            }
            // We don't know exact block indices, so use GetBlockData for known range.
            // Alternative: iterate by checking each block index until miss.
            // Since we don't have a ListBlockIndices method, iterate based on file size.
            size, found := bs.local.GetFileSize(ctx, payloadID)
            if !found {
                return
            }
            blockCount := size / blockstore.BlockSize
            if size % blockstore.BlockSize != 0 {
                blockCount++
            }
            for blockIdx := uint64(0); blockIdx < blockCount; blockIdx++ {
                if bs.readCache.Contains(payloadID, blockIdx) {
                    continue // already in L1
                }
                data, dataSize, err := bs.local.GetBlockData(ctx, payloadID, blockIdx)
                if err == nil && data != nil {
                    bs.readCache.Put(payloadID, blockIdx, data, dataSize)
                }
            }
            break
        }
    }
    ```

    Note: The auto-promote approach reads from local disk, which is in OS page cache after flush -- essentially zero-cost I/O. This is simpler than modifying the flush interface to return data (per RESEARCH.md recommendation).
