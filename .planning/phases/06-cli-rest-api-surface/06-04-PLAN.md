    func (rl RepoList) Headers() []string {
        return []string{"NAME", "KIND", "SCHEDULE", "RETENTION", "ENCRYPTED"}
    }

    func (rl RepoList) Rows() [][]string {
        out := make([][]string, 0, len(rl))
        for _, r := range rl {
            schedule := "-"
            if r.Schedule != nil && *r.Schedule != "" { schedule = *r.Schedule }
            out = append(out, []string{r.Name, r.Kind, schedule, renderRetention(r), renderEncrypted(r)})
        }
        return out
    }

    // renderRetention per D-20: "count=7 age=14d" / "count=7" / "age=14d" / "-"
    func renderRetention(r apiclient.BackupRepo) string {
        parts := []string{}
        if r.KeepCount != nil && *r.KeepCount > 0 { parts = append(parts, fmt.Sprintf("count=%d", *r.KeepCount)) }
        if r.KeepAgeDays != nil && *r.KeepAgeDays > 0 { parts = append(parts, fmt.Sprintf("age=%dd", *r.KeepAgeDays)) }
        if len(parts) == 0 { return "-" }
        return strings.Join(parts, " ")
    }

    func renderEncrypted(r apiclient.BackupRepo) string {
        if r.EncryptionEnabled { return "yes" }
        return "no"
    }

    func (d RepoDetail) Headers() []string { return []string{"FIELD", "VALUE"} }
    func (d RepoDetail) Rows() [][]string {
        rows := [][]string{
            {"Name", d.r.Name},
            {"Kind", d.r.Kind},
            {"Target", fmt.Sprintf("%s/%s", d.r.TargetKind, d.r.TargetID)},
        }
        if d.r.Schedule != nil { rows = append(rows, []string{"Schedule", *d.r.Schedule}) } else { rows = append(rows, []string{"Schedule", "-"}) }
        rows = append(rows, []string{"Retention", renderRetention(*d.r)})
        rows = append(rows, []string{"Encrypted", renderEncrypted(*d.r)})
        if d.r.EncryptionKeyRef != "" { rows = append(rows, []string{"KeyRef", d.r.EncryptionKeyRef}) }
        rows = append(rows, []string{"Created", d.r.CreatedAt.Format("2006-01-02 15:04:05")})
        rows = append(rows, []string{"Updated", d.r.UpdatedAt.Format("2006-01-02 15:04:05")})
        // Config: summarize — for local kind show path; for s3 show bucket/region/endpoint (omit secrets)
        rows = append(rows, renderConfigRows(d.r)...)
        return rows
    }

    func renderConfigRows(r *apiclient.BackupRepo) [][]string {
        rows := [][]string{}
        switch r.Kind {
        case "local":
            if v, ok := r.Config["path"]; ok { rows = append(rows, []string{"Path", fmt.Sprint(v)}) }
        case "s3":
            for _, k := range []string{"bucket", "region", "endpoint", "prefix"} {
                if v, ok := r.Config[k]; ok { rows = append(rows, []string{strings.Title(k), fmt.Sprint(v)}) }
            }
            // NEVER render secret_access_key / access_key_id values — mask as "***" if present
            if _, ok := r.Config["access_key_id"]; ok { rows = append(rows, []string{"AccessKeyID", "***"}) }
            if _, ok := r.Config["secret_access_key"]; ok { rows = append(rows, []string{"SecretAccessKey", "***"}) }
        }
        return rows
    }

    func buildRepoConfig(kind, jsonConfig, path, bucket, region, endpoint, prefix, accessKeyID, secretAccessKey string) (map[string]any, error) {
        if jsonConfig != "" {
            // Support @path/to/file.json
            raw := jsonConfig
            if strings.HasPrefix(raw, "@") {
                data, err := os.ReadFile(raw[1:])
                if err != nil { return nil, fmt.Errorf("read config file: %w", err) }
                raw = string(data)
            }
            var cfg map[string]any
            if err := json.Unmarshal([]byte(raw), &cfg); err != nil {
                return nil, fmt.Errorf("invalid JSON config: %w", err)
            }
            return cfg, nil
        }
        switch kind {
        case "local":
            p := path
            if p == "" {
                var err error
                p, err = prompt.InputRequired("Local backup path")
                if err != nil { return nil, cmdutil.HandleAbort(err) }
            }
            return map[string]any{"path": p}, nil
        case "s3":
            return buildS3Config(bucket, region, endpoint, prefix, accessKeyID, secretAccessKey)
        default:
            return nil, fmt.Errorf("unknown kind: %s (supported: local, s3)", kind)
        }
    }

    func buildS3Config(bucket, region, endpoint, prefix, accessKeyID, secretAccessKey string) (map[string]any, error) {
        // Copy S3 prompt flow from block/remote/add.go:112-170 verbatim, using the same
        // prompt.Input / prompt.Password helpers. Required: bucket, region. Optional:
        // endpoint, prefix. Credentials: D-15 research flag — implementer inspects
        // pkg/blockstore/remote/s3 for the canonical shape. Conservative default:
        // prompt for access_key_id + secret_access_key when not provided via flags.
        // If the implementer determines ambient AWS chain is sufficient, skip the
        // credential prompts and document the finding in the SUMMARY.
        cfg := map[string]any{}
        var err error
        if bucket == "" {
            bucket, err = prompt.InputRequired("S3 bucket")
            if err != nil { return nil, cmdutil.HandleAbort(err) }
        }
        cfg["bucket"] = bucket
        if region == "" {
            region, err = prompt.InputRequired("S3 region (e.g., us-east-1)")
            if err != nil { return nil, cmdutil.HandleAbort(err) }
        }
        cfg["region"] = region
        if endpoint != "" { cfg["endpoint"] = endpoint } else {
            epOpt, _ := prompt.InputOptional("S3 endpoint URL (leave empty for default)")
            if epOpt != "" { cfg["endpoint"] = epOpt }
        }
        if prefix != "" { cfg["prefix"] = prefix } else {
            pfxOpt, _ := prompt.InputOptional("S3 key prefix (leave empty for bucket root)")
            if pfxOpt != "" { cfg["prefix"] = pfxOpt }
        }
        // Credentials — D-15 research flag (implementer decides):
        if accessKeyID != "" { cfg["access_key_id"] = accessKeyID } else {
            akid, _ := prompt.InputOptional("S3 access key ID (leave empty for ambient AWS chain)")
            if akid != "" { cfg["access_key_id"] = akid }
        }
        if secretAccessKey != "" { cfg["secret_access_key"] = secretAccessKey } else if _, ok := cfg["access_key_id"]; ok {

            sk, err := prompt.Password("S3 secret access key")
            if err != nil { return nil, cmdutil.HandleAbort(err) }
            cfg["secret_access_key"] = sk
        }
        return cfg, nil
    }

    func runAdd(cmd *cobra.Command, args []string) error {
        storeName := args[0]
        client, err := cmdutil.GetAuthenticatedClient()
        if err != nil { return err }

        cfg, err := buildRepoConfig(addKind, addConfigJSON, addPath, addBucket, addRegion, addEndpoint, addPrefix, addAccessKeyID, addSecretAccessKey)
        if err != nil { return err }

        req := &apiclient.BackupRepoRequest{
            Name:   addName,
            Kind:   addKind,
            Config: cfg,
        }
        if addSchedule != "" { req.Schedule = &addSchedule }
        if addKeepCount > 0  { req.KeepCount = &addKeepCount }
        if addKeepAgeDays > 0 { req.KeepAgeDays = &addKeepAgeDays }
        if addEncryption != "" {
            on := addEncryption == "on"
            if on && addEncryptionKeyRef == "" {
                return fmt.Errorf("--encryption on requires --encryption-key-ref (env:KEY or file:/path)")
            }
            req.EncryptionEnabled = &on
        }
        if addEncryptionKeyRef != "" { req.EncryptionKeyRef = &addEncryptionKeyRef }

        repo, err := client.CreateBackupRepo(storeName, req)
        if err != nil {
            // Sentinel mapping: if it's an ErrScheduleInvalid from the server (400), propagate the message.
            return fmt.Errorf("failed to create backup repo: %w", err)
        }
        return cmdutil.PrintResourceWithSuccess(os.Stdout, repo, fmt.Sprintf("Backup repo '%s' created on store '%s'.", repo.Name, storeName))
    }

    var (
        removePurgeArchives bool
        removeForce         bool
    )

    var removeCmd = &cobra.Command{
        Use:   "remove <store-name> <repo-name>",
        Short: "Remove a backup repo from a metadata store",
        Long: `Remove a backup repo.

By default only the repo row (and its backup record rows) is deleted — the archive
files in the destination (manifest.yaml + payloads) remain on disk/S3 and can be
recovered by re-adding the repo with the same name.

With --purge-archives, the archive files are ALSO deleted from the destination.
This is irreversible.

Examples:
  # Default: config-only delete
  dfsctl store metadata fast-meta repo remove daily-s3

  # Also delete destination artifacts
  dfsctl store metadata fast-meta repo remove daily-s3 --purge-archives --force`,
        Args: cobra.ExactArgs(2),
        RunE: runRemove,
    }

    func init() {
        removeCmd.Flags().BoolVar(&removePurgeArchives, "purge-archives", false, "Also delete archive files in the destination (irreversible)")
        removeCmd.Flags().BoolVarP(&removeForce, "force", "f", false, "Skip confirmation prompt")
    }

    func runRemove(cmd *cobra.Command, args []string) error {
        storeName, repoName := args[0], args[1]
        client, err := cmdutil.GetAuthenticatedClient()
        if err != nil { return err }

        confirmationLabel := fmt.Sprintf("Backup repo '%s'", repoName)
        if removePurgeArchives {
            confirmationLabel += " (WILL ALSO DELETE ARCHIVE FILES)"
        }

        return cmdutil.RunDeleteWithConfirmation(confirmationLabel, repoName, removeForce, func() error {
            if err := client.DeleteBackupRepo(storeName, repoName, removePurgeArchives); err != nil {
                return fmt.Errorf("failed to delete backup repo: %w", err)
            }
            if removePurgeArchives {
                fmt.Printf("Backup repo '%s' removed (archive files purged).\n", repoName)
            } else {
                fmt.Printf("Backup repo '%s' removed (archive files retained).\n", repoName)
            }
            return nil
        })
    }
