    ```go
    package backup

    import (
        "context"
        "errors"
        "fmt"
        "os"
        "os/signal"
        "time"

        "github.com/marmos91/dittofs/pkg/apiclient"
        "github.com/marmos91/dittofs/internal/cli/output"
    )

    // Terminal statuses per Phase-1 enum + D-06 exit-code taxonomy
    var terminalStatuses = map[string]bool{
        "succeeded":   true,
        "failed":      true,
        "interrupted": true,
    }

    var (
        ErrPollTimeout   = errors.New("wait timeout")   // maps to exit 2 (D-11)
        ErrPollDetached  = errors.New("detached")       // user chose [d]etach; maps to exit 0 (D-08 rationale)
    )

    // WaitOptions configures WaitForJob for run.go / restore.go / job cancel + wait composition.
    type WaitOptions struct {
        StoreName string
        JobID     string
        Format    output.Format
        Timeout   time.Duration // 0 = indefinite (D-04)
    }

    // WaitForJob polls until terminal or user detach/cancel/timeout. Matches D-01..D-11.
    // Returns (finalJob, err). err interpretation:
    //   nil            — terminal reached; inspect finalJob.Status for succeeded/failed/interrupted
    //   ErrPollTimeout — --timeout expired; CLI exits 2, job continues server-side
    //   ErrPollDetached — user detached via Ctrl-C option [d]; CLI exits 0 (D-08)
    //   other          — transport or API error; exit 1
    func WaitForJob(ctx context.Context, client *apiclient.Client, opts WaitOptions) (*apiclient.BackupJob, error) {
        pollCtx, stop := signal.NotifyContext(ctx, os.Interrupt)
        defer stop()

        ticker := time.NewTicker(1 * time.Second) // D-05
        defer ticker.Stop()

        var deadline <-chan time.Time
        if opts.Timeout > 0 {
            deadline = time.After(opts.Timeout)
        }

        var lastStatus string
        spinner := newSpinner(opts.Format) // no-op in JSON/YAML mode per D-07

        for {
            select {
            case <-pollCtx.Done():
                if err := pollCtx.Err(); errors.Is(err, context.Canceled) {
                    action := handleCtrlC(opts.StoreName, opts.JobID)
                    switch action {
                    case "d":
                        fmt.Fprintf(os.Stderr, "Detached — job still running. Poll: dfsctl store metadata %s backup job show %s\n", opts.StoreName, opts.JobID)
                        return nil, ErrPollDetached
                    case "c":
                        if _, err := client.CancelBackupJob(opts.StoreName, opts.JobID); err != nil {
                            fmt.Fprintf(os.Stderr, "Cancel failed: %v (continuing to poll)\n", err)
                        }
                        stop()
                        pollCtx, stop = signal.NotifyContext(ctx, os.Interrupt)
                    case "C":
                        stop()
                        pollCtx, stop = signal.NotifyContext(ctx, os.Interrupt)
                    }
                }
            case <-deadline:
                fmt.Fprintf(os.Stderr, "Timeout — job still running. Poll: dfsctl store metadata %s backup job show %s\n", opts.StoreName, opts.JobID)
                return nil, ErrPollTimeout
            case <-ticker.C:
                job, err := client.GetBackupJob(opts.StoreName, opts.JobID)
                if err != nil {
                    return nil, err
                }
                if job.Status != lastStatus {
                    spinner.Transition(lastStatus, job.Status) // D-02 log transitions
                    lastStatus = job.Status
                }
                spinner.Tick()
                if terminalStatuses[job.Status] {
                    spinner.Stop()
                    return job, nil
                }
            }
        }
    }

    // handleCtrlC prompts "[d]etach (default) / [c]ancel / [C]ontinue" (D-03).
    func handleCtrlC(storeName, jobID string) string {
        fmt.Fprint(os.Stderr, "\nInterrupted. [d]etach (default) / [c]ancel / [C]ontinue watching: ")
        var response string
        fmt.Scanln(&response)
        switch response {
        case "c": return "c"
        case "C": return "C"
        default: return "d" // empty or anything else → detach per D-03 default
        }
    }

    // SuccessExitCode maps a terminal status to an exit code (D-06).
    func SuccessExitCode(status string) int {
        if status == "succeeded" { return 0 }
        return 1 // failed | interrupted
    }
    ```

    ```go
    package backup

    import (
        "fmt"
        "strings"
        "time"
    )

    // shortULID returns the first 8 chars of a ULID followed by "…" (D-26 discretion).
    func shortULID(id string) string {
        const prefixLen = 8
        if len(id) <= prefixLen { return id }
        return id[:prefixLen] + "…"
    }

    // timeAgo renders a relative duration such as "3h ago", "2d ago", "30s ago" (D-26).
    func timeAgo(t time.Time) string {
        d := time.Since(t)
        switch {
        case d < time.Minute: return fmt.Sprintf("%ds ago", int(d.Seconds()))
        case d < time.Hour:   return fmt.Sprintf("%dm ago", int(d.Minutes()))
        case d < 24*time.Hour: return fmt.Sprintf("%dh ago", int(d.Hours()))
        default:              return fmt.Sprintf("%dd ago", int(d.Hours()/24))
        }
    }

    // humanSize renders bytes as "12.3MB" / "1.0GB" / "234KB".
    func humanSize(b int64) string {
        const unit = 1024
        if b < unit { return fmt.Sprintf("%dB", b) }
        div, exp := int64(unit), 0
        for n := b / unit; n >= unit; n /= unit { div *= unit; exp++ }
        return fmt.Sprintf("%.1f%cB", float64(b)/float64(div), "KMGTPE"[exp])
    }

    // renderProgressBar renders a 20-char bar for D-47.
    func renderProgressBar(pct int) string {
        const width = 20
        if pct < 0 { pct = 0 }
        if pct > 100 { pct = 100 }
        filled := pct * width / 100
        return fmt.Sprintf("%d%%  [%s%s]", pct,
            strings.Repeat("▓", filled),
            strings.Repeat("░", width-filled))
    }
    ```

    ```go
    var (
        runRepo    string
        runWait    bool
        runAsync   bool
        runTimeout time.Duration
    )

    func registerRunFlags(c *cobra.Command) {
        c.Flags().StringVar(&runRepo, "repo", "", "Backup repo name (required when >1 repo attached per D-24)")
        c.Flags().BoolVar(&runWait, "wait", true, "Block until the backup job terminates (D-01 default)")
        c.Flags().BoolVar(&runAsync, "async", false, "Return immediately with the job record (D-01)")
        c.Flags().DurationVar(&runTimeout, "timeout", 0, "Wait timeout (0 = indefinite per D-04)")
    }

    func runRun(cmd *cobra.Command, args []string) error {
        if len(args) < 1 { return fmt.Errorf("store name required: dfsctl store metadata <name> backup") }
        storeName := args[0]
        client, err := cmdutil.GetAuthenticatedClient()
        if err != nil { return err }

        req := &apiclient.TriggerBackupRequest{Repo: runRepo}
        resp, err := client.TriggerBackup(storeName, req)
        if err != nil {
            var alreadyRunning *apiclient.BackupAlreadyRunningError
            if errors.As(err, &alreadyRunning) {
                fmt.Fprintf(os.Stderr, "Backup already running: %s.\n", alreadyRunning.RunningJobID)
                fmt.Fprintf(os.Stderr, "Show: dfsctl store metadata %s backup job show %s\n", storeName, alreadyRunning.RunningJobID)
                return fmt.Errorf("backup already running") // exit 1
            }
            return fmt.Errorf("failed to trigger backup: %w", err)
        }

        // resp.Record and resp.Job both populated in one round-trip (Plan 02 TriggerBackupResponse).
        if runAsync {
            return emitAsyncOutput(storeName, resp.Record, resp.Job) // D-09
        }

        // D-01 --wait default
        fmt.Fprintf(os.Stderr, "Backup job %s started (record %s). Polling...\n", resp.Job.ID, resp.Record.ID)
        job, err := WaitForJob(cmd.Context(), client, WaitOptions{
            StoreName: storeName,
            JobID:     resp.Job.ID,
            Timeout:   runTimeout,
            Format:    cmdutil.GetOutputFormatEnum(),
        })
        // D-08 detach: stderr-only message + stdout clean + exit 0 (user-initiated voluntary stop; job not failed).
        if errors.Is(err, ErrPollDetached) { return nil }
        // D-11 timeout: exit 2 (distinct from failure).
        if errors.Is(err, ErrPollTimeout)  { os.Exit(2); return nil }
        if err != nil                       { return err }

        // D-10 success output
        if cmdutil.GetOutputFormat() == "json" { return cmdutil.PrintResource(os.Stdout, job, nil) }
        fmt.Printf("✓ Backup completed\n")
        fmt.Printf("  Record:    %s\n", resp.Record.ID)
        fmt.Printf("  Duration:  %s\n", renderDuration(job))
        fmt.Printf("  Size:      %s\n", humanSize(resp.Record.SizeBytes))
        os.Exit(SuccessExitCode(job.Status))
        return nil
    }
    ```
