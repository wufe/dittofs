<objective>
Foundation layer for Phase 6. This plan does FOUR things:

1. Extend `BackupStore` with 4 new methods (`ListBackupRecords`,
   `ListBackupJobsFiltered`, `UpdateBackupRecordPinned`,
   `UpdateBackupJobProgress`), implement them in the GORM store, and
   conformance-test them.

2. Amend three runtime entrypoints so Phase-6 handlers + CLI can observe jobs:
   - `Service.RunBackup` returns `(*BackupRecord, *BackupJob, error)` (currently
     returns `(*BackupRecord, error)`) so the REST handler can hand back the
     polling BackupJob without a follow-up query.
   - `Service.RunRestore` returns `(*BackupJob, error)` (currently returns
     `error`) — same reason.
   - Add `Service.CancelBackupJob(ctx, jobID string) error` which looks up the
     run-ctx registered by `RunBackup` / `RunRestore` and invokes its
     `context.CancelFunc`. Idempotent on terminal jobs.
   - Add `Service.RunRestoreDryRun(ctx, repoID, recordID *string)
     (*DryRunResult, error)` — pre-flight (record resolution + manifest fetch
     + validation) without the shares-enabled gate, per D-31.
   These are tight adapter hooks over the existing async goroutine — NOT new
   business logic.

3. Instrument the backup + restore executors with the D-50 progress
   milestones (0/10/50/95 for backup; 0/10/30/60/95 for restore).

4. Surface `Enabled bool` on `apiclient.Share` with `json:"enabled"` so
   Plan 03 CLI `share list` / `share show` can render the column.
   (`models.Share` already has `json:"enabled"` — Phase 5 Plan 01 landed it;
   this plan only adds a regression test to lock that.)

Purpose: REST handlers (Plan 02) and CLI commands (Plans 03–06) cannot be
written until the store interface contract, cancel primitive, return-shape
amendments, and progress instrumentation are in place.

instrumentation + apiclient Share.Enabled field + conformance/regression
tests. No schema migration; no new business logic — the async goroutine
shape in RunBackup / RunRestore is preserved, the BackupJob row is already
created synchronously before the spawned goroutine, so returning its handle
to the caller is a straight refactor.

    ```go
    func (s *GORMStore) ListBackupRecords(ctx context.Context, repoID string, statusFilter models.BackupStatus) ([]*models.BackupRecord, error) {
        q := s.db.WithContext(ctx).Where("repo_id = ?", repoID)
        if statusFilter != "" {
            q = q.Where("status = ?", statusFilter)
        }
        var results []*models.BackupRecord
        if err := q.Order("created_at DESC").Find(&results).Error; err != nil {
            return nil, err
        }
        return results, nil
    }

    func (s *GORMStore) ListBackupJobsFiltered(ctx context.Context, filter BackupJobFilter) ([]*models.BackupJob, error) {
        q := s.db.WithContext(ctx)
        if filter.RepoID != "" { q = q.Where("repo_id = ?", filter.RepoID) }
        if filter.Status != "" { q = q.Where("status = ?", filter.Status) }
        if filter.Kind   != "" { q = q.Where("kind = ?",   filter.Kind) }
        limit := filter.Limit
        if limit <= 0 { limit = 50 }
        if limit > 200 { limit = 200 }
        q = q.Order("started_at IS NULL, started_at DESC").Limit(limit) // sqlite-safe NULLS LAST
        var results []*models.BackupJob
        if err := q.Find(&results).Error; err != nil { return nil, err }
        return results, nil
    }

    func (s *GORMStore) UpdateBackupRecordPinned(ctx context.Context, recordID string, pinned bool) error {
        result := s.db.WithContext(ctx).
            Model(&models.BackupRecord{}).
            Where("id = ?", recordID).
            Update("pinned", pinned)
        if result.Error != nil { return result.Error }
        if result.RowsAffected == 0 { return models.ErrBackupRecordNotFound }
        return nil
    ```

    ```go
    // CancelBackupJob cancels the in-flight executor run-ctx for jobID.
    // Returns ErrBackupJobNotFound if the job has no registered run-ctx
    // (unknown ID or already terminal — the REST handler maps the latter
    // to 200 OK + current job per D-45 idempotent semantics).
    //
    // Cancellation is synchronous w.r.t. ctx propagation: the run-ctx
    // is canceled before this call returns. The executor goroutine observes
    // ctx.Done() and winds down its own cleanup path (writing the final
    // BackupJob row with Status=interrupted). Waiting for the goroutine
    // to finish is the caller's responsibility (Phase 4 D-18 path).
    func (s *Service) CancelBackupJob(ctx context.Context, jobID string) error {
        s.runCtxRegistry.Lock()
        cancel, ok := s.runCtxRegistry.cancels[jobID]
        s.runCtxRegistry.Unlock()
        if !ok {
            return ErrBackupJobNotFound
        }
        cancel()
        return nil
    }
    ```

    ```go
    // DryRunResult is returned by Service.RunRestoreDryRun — pre-flight only,
    // no state mutation, no payload download (D-31).
    type DryRunResult struct {
        Record        *models.BackupRecord // selected record (latest succeeded OR explicit --from)
        ManifestValid bool                 // true if manifest_version + store_id + store_kind + sha256 all validated
        EnabledShares []string             // shares currently enabled on the target store (REPORTED for the CLI hint; NOT a refusal)
    }

    // RunRestoreDryRun performs the D-31 pre-flight: record resolution +
    // Destination.GetManifestOnly + manifest validation. Does NOT enforce
    // the shares-enabled gate (unlike RunRestore). Cheap — manifests are
    // KBs, not GBs.
    func (s *Service) RunRestoreDryRun(ctx context.Context, repoID string, recordID *string) (*DryRunResult, error) {
        if s.restoreExec == nil || s.shares == nil || s.stores == nil {
            return nil, fmt.Errorf("restore path not wired")
        }
        // Load repo + resolve target (no side effects)
        repo, err := s.store.GetBackupRepoByID(ctx, repoID)
        if err != nil {
            if errors.Is(err, models.ErrBackupRepoNotFound) {
                return nil, fmt.Errorf("%w: %s", ErrRepoNotFound, repoID)
            }
            return nil, fmt.Errorf("load repo: %w", err)
        }
        resolver, ok := s.resolver.(RestoreResolver)
        if !ok {
            return nil, fmt.Errorf("resolver does not implement RestoreResolver")
        }
        _, storeID, storeKind, storeName, err := resolver.ResolveWithName(ctx, repo.TargetKind, repo.TargetID)
        if err != nil { return nil, err }

        // Record selection (same logic as RunRestore — extract to a private helper
        // `selectRestoreRecord` that both paths call; it already exists in restore.go).
        selectedID, err := s.selectRestoreRecord(ctx, repoID, recordID)
        if err != nil { return nil, err }
        rec, err := s.store.GetBackupRecordByID(ctx, selectedID)
        if err != nil { return nil, err }

        // Build destination and fetch manifest only (cheap)
        dst, err := s.destFactory(ctx, repo)
        if err != nil { return nil, fmt.Errorf("build destination: %w", err) }
        defer dst.Close()

        manifest, err := dst.GetManifestOnly(ctx, rec.ManifestPath)
        if err != nil { return nil, fmt.Errorf("fetch manifest: %w", err) }

        // Validate manifest envelope (version, store id, store kind, sha256 non-empty).
        // Delegate to the same validator the restore executor uses.
        manifestValid := true
        if verr := validateRestoreManifest(manifest, storeID, storeKind); verr != nil {
            // Do NOT return err — dry-run reports validity as false so CLI
            // can still show the selected record + why it won't be usable.
            // EXCEPTION: forward-incompatible version IS fatal; propagate so
            // the CLI does not proceed with an unusable record selection.
            if errors.Is(verr, ErrManifestVersionUnsupported) { return nil, verr }
            manifestValid = false
        }

        // REPORT enabled shares but do NOT gate (D-31).
        enabled := s.shares.ListEnabledSharesForStore(storeName)

        return &DryRunResult{
            Record:        rec,
            ManifestValid: manifestValid,
            EnabledShares: enabled,
        }, nil
    }
    ```
