<objective>
Implement eviction policy enforcement in FSStore so that pinned blocks are never evicted, TTL blocks are only evicted after their file's last-access time exceeds the threshold, and LRU blocks are evicted in true last-access order.

Purpose: This is the core behavior that makes cache retention meaningful. Without eviction enforcement, the model fields from plan 01 are just data — this plan makes them control actual eviction decisions.

Output: Policy-aware ensureSpace(), per-file access time tracker, eviction unit tests, and wiring from share creation to FSStore config.
</objective>

3. Rewrite `pkg/blockstore/local/fs/eviction.go` `ensureSpace()`:
   ```go
   func (bc *FSStore) ensureSpace(ctx context.Context, needed int64) error {
       if bc.maxDisk <= 0 {
           return nil
       }

       // Pin mode: never evict, just check space
       if bc.retentionPolicy == blockstore.RetentionPin {
           if bc.diskUsed.Load()+needed > bc.maxDisk {
               return ErrDiskFull
           }
           return nil
       }

       // Eviction disabled (local-only mode) — same as current behavior
       if !bc.evictionEnabled.Load() {
           if bc.diskUsed.Load()+needed > bc.maxDisk {
               return ErrDiskFull
           }
           return nil
       }

       const maxWait = 30 * time.Second
       deadline := time.Now().Add(maxWait)
       recalculated := false

       for bc.diskUsed.Load()+needed > bc.maxDisk {
           evicted := false

           switch bc.retentionPolicy {
           case blockstore.RetentionTTL:
               evicted, _ = bc.evictTTLExpired(ctx)
           default: // LRU (default, also handles empty string)
               evicted, _ = bc.evictLRU(ctx)
           }

           if !evicted {
               if !recalculated {
                   recalculated = true
                   bc.recalcDiskUsed()
                   if bc.diskUsed.Load()+needed <= bc.maxDisk {
                       break
                   }
               }
               if time.Now().After(deadline) {
                   return ErrDiskFull
               }
               select {
               case <-ctx.Done():
                   return ctx.Err()
               case <-time.After(100 * time.Millisecond):
                   continue
               }
           }
       }

       return nil
   }
   ```

4. Add `evictTTLExpired()` method:
   - Call `ListRemoteBlocks(ctx, 0)` to get all evictable blocks (or a batch).
   - For each block, extract payloadID from block ID, look up file last-access time from accessTracker.
   - Filter: only blocks whose file last-access is older than `now - retentionTTL`.
   - Sort filtered by last-access (oldest first) for fairness among expired blocks.
   - Evict the oldest one.
   - Returns (evicted bool, error).

5. Add `evictLRU()` method:
   - Call `ListRemoteBlocks(ctx, 0)` to get all evictable blocks.
   - For each block, look up per-file last-access from accessTracker. Use the file-level access time (not block-level LastAccess) for true per-file LRU.
   - Sort by file last-access (oldest first).
   - Evict the oldest one.
   - Returns (evicted bool, error).
   - This replaces the current "evict first block found" approach with true LRU.
